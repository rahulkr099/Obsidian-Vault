### ðŸ“„ **Better SPA Example (Pure JS)**

```html
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8" />
    <meta name="viewport" content="width=device-width, initial-scale=1.0" />
    <title>Simple SPA</title>

    <style>
        body {
            font-family: Arial, sans-serif;
            padding: 1rem;
        }

        nav a {
            margin-right: 1rem;
            cursor: pointer;
            color: blue;
            text-decoration: underline;
        }
    </style>
</head>
<body>

    <nav>
        <a data-page="home">Home</a>
        <a data-page="about">About</a>
    </nav>

    <div id="content"></div>

    <script>
        const content = document.getElementById("content");

        function renderPage(page) {
            content.textContent = ""; // clear content

            const heading = document.createElement("h2");

            if (page === "home") {
                heading.textContent = "Home";
            } else if (page === "about") {
                heading.textContent = "About";
            } else {
                heading.textContent = "Page Not Found";
            }

            content.appendChild(heading);i
        }

        document.querySelector("nav").addEventListener("click", function (event) {
            if (event.target.dataset.page) {
                renderPage(event.target.dataset.page);
            }
        });

        // Load default page
        renderPage("home");
    </script>

</body>
</html>
```
