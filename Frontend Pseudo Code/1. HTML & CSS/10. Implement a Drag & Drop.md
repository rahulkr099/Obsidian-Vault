## 2ï¸âƒ£ Cleaner, Less-Direct JavaScript (Beginner Friendly)


```javascript
    let draggedItemId = null;

    function allowDrop(event) {
        // Step 1: Allow dropping
        event.preventDefault();
    }

    function drag(event) {
        // Step 2: Get dragged element
        const draggedElement = event.target;

        // Step 3: Store its id
        draggedItemId = draggedElement.id;

        // Step 4: Save id in drag memory
        event.dataTransfer.setData("text/plain", draggedItemId);
    }

    function drop(event) {
        // Step 5: Stop default browser behavior
        event.preventDefault();

        // Step 6: Identify drop area
        const dropArea = event.target;

        // Step 7: Get dragged item id
        const receivedItemId = event.dataTransfer.getData("text/plain");

        // Step 8: Find dragged element
        const draggedElement = document.getElementById(receivedItemId);

        // Step 9: Move dragged element into drop area
        dropArea.appendChild(draggedElement);
    }

```

<!DOCTYPE html>

<div class="container">

    <div class="items">
        <div class="drag-item" draggable="true">Item 1</div>
        <div class="drag-item" draggable="true">Item 2</div>

<script>
    const dragItems = document.querySelectorAll(".drag-item");
    const dropZone = document.getElementById("dropZone");
    let draggedItem = null;

    // Drag start
    dragItems.forEach(item => {
        item.addEventListener("dragstart", e => {
            draggedItem = item;

            // Custom drag preview
            const preview = item.cloneNode(true);
            preview.style.position = "absolute";
            preview.style.top = "-1000px";
            document.body.appendChild(preview);
            e.dataTransfer.setDragImage(preview, 10, 10);

            setTimeout(() => document.body.removeChild(preview), 0);
        });
    });

    // Allow drop
    dropZone.addEventListener("dragover", e => {
        e.preventDefault();
        dropZone.classList.add("hover");
    });

    // Leave drop zone
    dropZone.addEventListener("dragleave", () => {
        dropZone.classList.remove("hover");
    });

    // Drop
    dropZone.addEventListener("drop", e => {
        e.preventDefault();
        dropZone.classList.remove("hover");

        if (draggedItem) {
            dropZone.appendChild(draggedItem);
            draggedItem = null;
        }
    });

    // ---------- MOBILE TOUCH SUPPORT ----------
    dragItems.forEach(item => {
        item.addEventListener("touchstart", e => {
            draggedItem = item;
        });

        item.addEventListener("touchend", e => {
            const touch = e.changedTouches[0];
            const dropRect = dropZone.getBoundingClientRect();

            if (
                touch.clientX > dropRect.left &&
                touch.clientX < dropRect.right &&
                touch.clientY > dropRect.top &&
                touch.clientY < dropRect.bottom
            ) {
                dropZone.appendChild(draggedItem);
            }

            draggedItem = null;
        });
    });
</script>
